\begin{align*}
    &\bar{1}, \bar{x}, \bar{x}^2, \ldots, \bar{x}^{d-1} \quad \text{basis for } L \\
    &\bar{1} = 1 + <f> \\ 
    &\bar{x} = x + <f> \text{ spn L over K} \\
    &k(x) / <f> \text{ is isomorphic to } L \\
\end{align*}
